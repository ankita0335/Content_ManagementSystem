CREATE DATABASE CMS;
USE CMS;
CREATE TABLE CONTENT (
C_ID VARCHAR(100) NOT NULL PRIMARY KEY,
C_TITLE VARCHAR(255),
CURR_DATE DATETIME,
C_IMAGE VARCHAR(250) NOT NULL,
C_AUTHER VARCHAR(255),
C_ARTICLE LONGTEXT character SET utf8mb4 collate utf8mb4_unicode_ci;
);



CREATE TABLE REGISTER (
FULL_NAME VARCHAR(50),
EMAIL VARCHAR(100) NOT NULL PRIMARY KEY,
DOB DATE,
PHONE BIGINT,
USER_TYPE VARCHAR(20),
PASS VARCHAR(30) NOT NULL
);
